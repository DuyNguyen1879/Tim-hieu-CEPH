[root@vinhvt-ceph1 my-cluster]# ceph-deploy install node1 node2 node3

```
1: Chạy yum clean all  (remove cached data)

2. yum -y install epel-release (install epel repo)

3. yum -y install yum-plugin-priorities (extensions điều chỉnh propities cho repo)

4. rpm --import https://download.ceph.com/keys/release.asc

5. yum remove -y ceph-release (remove repo ceph cux)

6. yum install -y https://download.ceph.com/rpm-mimic/el7/noarch/ceph-release-1-0.el7.noarch.rpm (cài dặt repo ceph moiws)

7. yum -y install ceph ceph-radosgw (cài đặt ceph và ceph-radosgw)
```

[root@admin my-cluster]# ceph-deploy mon create-initial

```
1. Write cấu hình cluster vào /etc/ceph/{cluster}.conf

2. Tao keyring file: /var/lib/ceph/tmp/ceph-node1.mon.keyring

2. Khoi tạo thu muc mondata trên node1 voi keyring ceph-node1.mon.keyring bằng câu lệnh: ceph-mon --cluster ceph --mkfs -i node1 --keyring /var/lib/ceph/tmp/ceph-node1.mon.keyring --setuser 167 --setgroup 167

3. Unlinking keyring file /var/lib/ceph/tmp/ceph-node1.mon.keyring

3. systemctl enable ceph.target

4. systemctl enable ceph-mon@node1

6. systemctl start ceph-mon@node1

7. Check status mon : ceph --cluster=ceph --admin-daemon /var/run/ceph/ceph-mon.node1.asok mon_status

8. Export các key vào client
```

[root@admin my-cluster]# ceph-deploy osd create --data /dev/vdb node1

```
1. Tạo pv, vg, lv

2. add keyring vao /var/lib/ceph/osd/ceph-0/keyring

3. ceph-volume prepare succesfull

4. systemctl enable --runtime ceph-osd@0

5. systemctl start ceph-osd@0

6. ceph --cluster=ceph osd stat --format=json
```

[root@admin my-cluster]# ceph-deploy mgr create node2

```
1. Tao keyring file : ceph --cluster ceph --name client.bootstrap-mgr --keyring /var/lib/ceph/bootstrap-mgr/ceph.keyring auth get-or-create mgr.node2 mon allow profile mgr osd allow * mds allow * -o /var/lib/ceph/mgr/ceph-node2/keyring ( cho phep monnitor ket noi den mgr va co full quyen voi osd,mds )

2. systemctl enable ceph-mgr@node2

3. systemctl start ceph-mgr@node2

4. systemctl enable ceph.target
```
